// Kayıt kısmındaki tıklama fonksiyonunu güncelliyoruz canvas.addEventListener('click', (event) => { if (!isImageLoaded) return; // Kullanıcıya bu noktanın ne olduğunu soralım (Basit bir prompt) const label = prompt("Bu nokta neyi temsil ediyor? (Örn: Göz-Mavi, Agiz-Kirisik)", "Göz-Mavi"); if (!label) return; const rect = canvas.getBoundingClientRect(); const x = (event.clientX - rect.left) * (canvas.width / rect.width); const y = (event.clientY - rect.top) * (canvas.height / rect.height); selectedPoints.push({ x, y, label }); // Artık 'label'ı da kaydediyoruz! // Görsel geri bildirim (kırmızı daire) ctx.fillStyle = '#ff0000'; ctx.beginPath(); ctx.arc(x, y, 5, 0, Math.PI * 2); ctx.fill(); ctx.fillText(label, x + 10, y); // Ekrana etiketi yaz });
